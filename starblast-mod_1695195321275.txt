function r() {
  var r_value = Math.random();
  var id;
  if (r_value <= 0.2) {
    id = 119;
  } else if ((r_value > 0.2) && (r_value <= 0.4)) {
    id = 161;
  } else if ((r_value > 0.4) && (r_value <= 0.6)) {
    id = 201;
  } else if ((r_value > 0.6) && (r_value <= 0.8)) {
    id = 240;
  } else {
    id = 244;
  }
  return id;
}

          ]

        ship.custom.option_screen = true;    
        ship.custom.tree = 0;
        for (let tree = 0; tree < ship_list.length; tree++){
          if (ship_list[tree].indexOf(ship.type) >= 0){
            ship.custom.tree = tree;

          }
        }        
      }
    }
    if (game.aliens.length < max_aliens && alien_portal === true){
      game.addAlien(alien_array[Math.floor(Math.random()*alien_array.length)]);
    }
    if (alien_portal === false){

    }
  }
  if (game.step % 1800 === 0 && game.ships.length > 1) playerList();
};

          shortcut: `${shc[i]}`,
          components: [
            {type:"box",position:[0,0,100,100],fill:"rgba(68, 85, 102, 0)",stroke:"#CDE",width:4},
            {type: "text",position:[5,35,90,40],value:`${ids[i]} [${i+1}]`,color:"#CDE"},
          ]
        });      }  
      ship.custom.option_screen = false;
    } else if (ship.custom.option_screen === false){
      for (let i=0; i<ids.length; i++) sendUI(ship, {id:ids[i],visible:false});
      ship.custom.option_screen = true;
    }    
  }
};

this.event = function(event, game){
  var ship = event.ship 
  switch (event.name){
    case "ui_component_clicked":
      var component = event.id;
      switch (component){
        case "Options":
          UIevents.options(ship);          
          break;  
        case "Reset":
          UIevents.reset(ship);          
          break;  
        }
      break;
    case "ship_spawned": 
      ship.set({x:0,y:250,invulnerable:300,crystals:((Math.trunc(ship.type/100)||0)**2)*20});
    break;
  }
};
